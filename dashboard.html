Scatter Plot with color based on order_hour_of_day

line_chart <- sample_data |>
  mutate(time_of_day = case_when(
    order_hour_of_day < 12 ~ "Morning",
    order_hour_of_day < 18 ~ "Afternoon",
    TRUE ~ "Evening")
  ) |>
  group_by(order_dow, time_of_day) |>
  summarise(mean_days_since_prior = mean(days_since_prior_order, na.rm = TRUE)) |>
  plot_ly(
    x = ~order_dow, 
    y = ~mean_days_since_prior,  
    color = ~time_of_day, 
    type = "scatter", 
    mode = "markers",
    line = list(shape = "spline")  # Optional: This makes the lines curved instead of straight
  )
